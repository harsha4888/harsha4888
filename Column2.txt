Column 2 = 
VAR CurrentQuarter = MAX('Viz Expenses'[quater])  -- Get the current (latest) quarter in 'YYQX' format
VAR CurrentYear = LEFT(CurrentQuarter, 2)         -- Extract the 'YY' part from 'YYQX'
VAR CurrentQtr = MID(CurrentQuarter, 4, 1)        -- Extract the 'Q' part from 'YYQX'

-- Calculate the previous quarter (Last Quarter)
VAR LastQuarter =
    IF(
        CurrentQtr = "1", 
        FORMAT(VALUE(CurrentYear) - 1, "00") & "Q4",  -- If it's Q1, subtract 1 from the year and set the quarter to Q4
        CurrentYear & "Q" & FORMAT(VALUE(CurrentQtr) - 1, "0")  -- For other quarters, subtract 1 from the quarter
    )
RETURN
    IF('Viz Expenses'[quater] = CurrentQuarter, "Current Quarter",
        IF('Viz Expenses'[quater] = LastQuarter, "Last Quarter", "")
    )
