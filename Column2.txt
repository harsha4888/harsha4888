Column2 =
VAR MaxMonth = CALCULATE(MAX(MONTH('Viz Payments'[DatePay])), ALL('Viz Payments'))
VAR MaxYear = CALCULATE(MAX(YEAR('Viz Payments'[DatePay])), ALL('Viz Payments'))
VAR CurrentMonth = MONTH(TODAY())
VAR CurrentYear = YEAR(TODAY())
VAR LastMonth = IF(CurrentMonth = 1, 12, CurrentMonth - 1)
VAR LastMonthYear = IF(CurrentMonth = 1, CurrentYear - 1, CurrentYear)
VAR IsCurrentMonth = YEAR('Viz Payments'[DatePay]) = CurrentYear 
                      && MONTH('Viz Payments'[DatePay]) = MaxMonth
VAR IsLastMonth = YEAR('Viz Payments'[DatePay]) = LastMonthYear 
                   && MONTH('Viz Payments'[DatePay]) = LastMonth
                   || YEAR('Viz Payments'[DatePay]) = CurrentYear - 1 
                   && MaxMonth = 1 
                   && MONTH('Viz Payments'[DatePay]) = 12

RETURN
IF(
    IsCurrentMonth,
    "Current Month",
    IF(
        IsLastMonth,
        "Last Month",
        "Older Month"
    )
)
