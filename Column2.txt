Column2 =
VAR MaxDate = CALCULATE(MAX('Viz Payments'[DatePay]), ALL('Viz Payments'))
VAR MaxMonth = MONTH(MaxDate)
VAR MaxYear = YEAR(MaxDate)
VAR LastMonth = IF(MaxMonth = 1, 12, MaxMonth - 1)
VAR LastMonthYear = IF(MaxMonth = 1, MaxYear - 1, MaxYear)

VAR DatePayYear = YEAR('Viz Payments'[DatePay])
VAR DatePayMonth = MONTH('Viz Payments'[DatePay])

VAR IsCurrentMonth = DatePayYear = MaxYear && DatePayMonth = MaxMonth
VAR IsLastMonth = DatePayYear = LastMonthYear && DatePayMonth = LastMonth

RETURN
IF(
    IsCurrentMonth,
    "Current Month",
    IF(
        IsLastMonth,
        "Last Month",
        "Older Month"
    )
)
