quater = 
VAR SelectedDate = 
    SWITCH(
        TRUE(),
        [date_filtered] = 0, MIN('Viz Expenses'[Created Date]),
        [date_filtered] = 1, MIN('Viz Expenses'[Payment Date]),
        MIN('Viz Expenses'[Transaction Date])
    )
RETURN
    RIGHT(YEAR(SelectedDate), 2) & "Q" & QUARTER(SelectedDate)
