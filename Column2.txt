Quart Label = 
VAR MaxQuart = MAXX(ALL('Viz Expenses'[Quart]), 'Viz Expenses'[Quart])
VAR LastQuart = MaxQuart - 1
RETURN
    IF(
        'Viz Expenses'[Quart] = MaxQuart, 
        "Current Quarter",
        IF(
            'Viz Expenses'[Quart] = LastQuart, 
            "Last Quarter", 
            ""
        )
    )
